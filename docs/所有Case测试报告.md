使用自动化测试脚本 `test_all_cases.sh`：

```bash
#!/bin/bash
for kb_dir in kbscript/*/; do
    [ -d "$kb_dir" ] || continue
    [ -f "$kb_dir/run.sh" ] || continue
    
    kb_name=$(basename "$kb_dir")
    export KB_RESULT_FILE="temp/test_${kb_name}.json"
    
    bash "$kb_dir/run.sh" 2>&1
    
    # 验证结果
    steps=$(jq '.steps | length' "$KB_RESULT_FILE")
    score=$(jq '.score' "$KB_RESULT_FILE")
done
```
