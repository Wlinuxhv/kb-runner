# KB脚本执行框架概要详细设计文档

## 1. 文档概述

### 1.1 文档目的
本文档为KB脚本执行框架提供详细的技术设计方案，指导开发团队进行系统实现。文档涵盖系统架构、模块设计、数据结构、接口定义、流程设计等核心内容。

### 1.2 参考文档
- 《KB脚本执行框架用户需求文档》
- 《KB脚本执行框架系统需求文档》

### 1.3 术语定义
| 术语 | 解释 |
|------|------|
| KB | Knowledge Base，知识库，包含检查逻辑和规则 |
| Runner | 脚本执行器，负责调度和执行脚本 |
| Adapter | 脚本适配器，适配不同脚本语言 |
| Result | 执行结果，包含步骤状态和输出信息 |
| Matrix | 结果矩阵，多个脚本执行结果的加权归一化矩阵 |

```go
// cmd/kb-runner/main.go
package main

import (
    "context"
    "fmt"
    "os"
    "time"
    
    "github.com/spf13/cobra"
    "kb-runnerx/internal/executor"
    "kb-runnerx/internal/adapter"
    "kb-runnerx/internal/processor"
    "kb-runnerx/internal/case"
    "kb-runnerx/internal/scenario"
    "kb-runnerx/pkg/config"
    "kb-runnerx/pkg/logger"
)

var (
    cfgFile     string
    scriptPath  string
    language    string
    params      []string
    timeout     int
    parallel    int
    outputDir   string
    outputFormat string
    verbose     bool
    quiet       bool
    weight      float64

    interactive bool

    tags        []string

var rootCmd = &cobra.Command{
    Use:   "kb-runner",
    Short: "KB脚本执行框架",
    Long: `KB脚本执行框架 - 用于执行KB检查脚本并生成结果矩阵

支持多种脚本语言（Bash、Python），提供标准化的结果格式和权重计算。

使用方式:
  kb-runner list              列出所有可用的CASE
  kb-runner show <case-name>  查看CASE详情
  kb-runner scenario list     列出所有场景
  kb-runner run [options]     执行脚本或CASE`,
}

var listCmd = &cobra.Command{
    Use:   "list",
    Short: "列出所有可用的CASE",
    Long: `列出所有已注册的CASE，支持按分类、标签筛选和搜索。

示例:
  kb-runner list
  kb-runner list --category security
  kb-runner list --tags critical,daily
  kb-runner list --search "check"`,
    RunE: listCases,
}

var showCmd = &cobra.Command{
    Use:   "show <case-name>",
    Short: "显示CASE详细信息",
    Long: `显示指定CASE的详细配置信息，包括脚本路径、参数、权重等。

示例:
  kb-runner show security_check`,
    Args: cobra.ExactArgs(1),
    RunE: showCase,
}

var scenarioCmd = &cobra.Command{
    Use:   "scenario",
    Short: "场景管理命令",
}

var scenarioListCmd = &cobra.Command{
    Use:   "list",
    Short: "列出所有场景",

var runCmd = &cobra.Command{

    Short: "执行KB脚本",
    Long: `执行一个或多个KB脚本，支持多种执行方式：

1. 直接执行脚本文件:
   kb-runner run -s ./scripts/check.sh -l bash

2. 按CASE名称执行:

4. 交互式选择执行:

var versionCmd = &cobra.Command{
    Use:   "version",
    Short: "显示版本信息",
    Run: func(cmd *cobra.Command, args []string) {

func init() {
    cobra.OnInitialize(initConfig)
    
    // 全局参数
    rootCmd.PersistentFlags().StringVarP(&cfgFile, "config", "c", "", "配置文件路径")
    rootCmd.PersistentFlags().BoolVarP(&verbose, "verbose", "v", false, "详细输出")
    rootCmd.PersistentFlags().BoolVarP(&quiet, "quiet", "q", false, "静默模式")
    
    // list命令参数
    listCmd.Flags().StringVarP(&category, "category", "C", "", "按分类筛选")
    listCmd.Flags().StringSliceVarP(&tags, "tags", "T", nil, "按标签筛选")
    listCmd.Flags().StringVarP(&searchTerm, "search", "S", "", "搜索CASE")
    listCmd.Flags().StringVarP(&outputFormat, "format", "f", "table", "输出格式")
    listCmd.Flags().BoolVarP(&verbose, "all", "a", false, "显示所有详细信息")
    
    // scenario命令
    scenarioCmd.AddCommand(scenarioListCmd)
    scenarioCmd.AddCommand(scenarioShowCmd)
    
    // run命令参数
    runCmd.Flags().StringVarP(&scriptPath, "script", "s", "", "脚本文件路径")
    runCmd.Flags().StringVarP(&language, "language", "l", "bash", "脚本语言类型")
    runCmd.Flags().StringArrayVarP(&params, "param", "p", nil, "脚本参数 (key=value)")
    runCmd.Flags().IntVarP(&timeout, "timeout", "t", 300, "执行超时时间(秒)")
    runCmd.Flags().IntVarP(&parallel, "parallel", "n", 1, "并行执行数量")
    runCmd.Flags().StringVarP(&outputDir, "output", "o", "", "结果输出目录")
    runCmd.Flags().StringVarP(&outputFormat, "format", "f", "json", "输出格式 (json/yaml/table)")
    runCmd.Flags().Float64VarP(&weight, "weight", "w", 1.0, "脚本权重")
    runCmd.Flags().StringVar(&caseName, "case", "", "按CASE名称执行")
    runCmd.Flags().StringVar(&scenarioName, "scenario", "", "按场景执行")
    runCmd.Flags().BoolVarP(&interactive, "interactive", "i", false, "交互式选择执行")
    runCmd.Flags().StringVar(&category, "category", "", "执行指定分类的所有CASE")
    runCmd.Flags().StringSliceVar(&tags, "tags", nil, "执行指定标签的所有CASE")
    
    rootCmd.AddCommand(listCmd)
    rootCmd.AddCommand(showCmd)
    rootCmd.AddCommand(scenarioCmd)
    rootCmd.AddCommand(runCmd)
    rootCmd.AddCommand(versionCmd)
}

func initConfig() {
    if cfgFile != "" {
        config.LoadFile(cfgFile)
    }
}

func listCases(cmd *cobra.Command, args []string) error {
    caseManager := case.NewManager(config.Load())
    cases, err := caseManager.List(case.FilterOptions{
        Category: category,
        Tags:     tags,
        Search:   searchTerm,
    })
    if err != nil {
        return err
    }
    return outputCaseList(cases, outputFormat, verbose)
}

func showCase(cmd *cobra.Command, args []string) error {
    caseName := args[0]
    caseManager := case.NewManager(config.Load())
    c, err := caseManager.Get(caseName)
    if err != nil {
        return err
    }
    return outputCaseDetail(c, outputFormat)
}

func listScenarios(cmd *cobra.Command, args []string) error {
    scenarioManager := scenario.NewManager(config.Load())
    scenarios, err := scenarioManager.List()
    if err != nil {
        return err
    }
    return outputScenarioList(scenarios, outputFormat)
}

func showScenario(cmd *cobra.Command, args []string) error {
    scenarioName := args[0]
    scenarioManager := scenario.NewManager(config.Load())
    s, err := scenarioManager.Get(scenarioName)
    if err != nil {
        return err
    }
    return outputScenarioDetail(s, outputFormat)
}

func runScripts(cmd *cobra.Command, args []string) error {
    ctx := context.Background()
    log := logger.New(verbose, quiet)
    cfg, err := config.Load()
    if err != nil {
        return fmt.Errorf("加载配置失败: %w", err)
    }
    
    var tasks []*executor.Task
    
    // 根据不同执行方式构建任务
    switch {
    case interactive:
        // 交互式选择
        tasks, err = interactiveSelect(cfg)
    case scenarioName != "":
        // 按场景执行
        tasks, err = buildTasksFromScenario(cfg, scenarioName)
    case caseName != "":
        // 按CASE名称执行
        tasks, err = buildTasksFromCase(cfg, caseName)
    case category != "":
        // 按分类执行
        tasks, err = buildTasksFromCategory(cfg, category)
    case len(tags) > 0:
        // 按标签执行
        tasks, err = buildTasksFromTags(cfg, tags)
    default:
        // 直接执行脚本
        tasks = buildTasks(args)
    }
    
    if err != nil {
        return err
    }
    
    engine := executor.NewEngine(cfg, log)
    results, err := engine.ExecuteBatch(ctx, tasks)
    if err != nil {
        return fmt.Errorf("执行失败: %w", err)
    }
    
    proc := processor.NewProcessor(cfg, log)
    matrix, err := proc.Process(results)
    if err != nil {
        return fmt.Errorf("结果处理失败: %w", err)
    }
    
    return outputResults(matrix, outputFormat, outputDir)
}

func main() {
    if err := rootCmd.Execute(); err != nil {
        fmt.Fprintln(os.Stderr, err)
        os.Exit(1)
    }
}
```

#### 4.2.2 命令参数详解

| 参数 | 类型 | 默认值 | 说明 |
|------|------|--------|------|
| `-c, --config` | string | - | 配置文件路径 |
| `-s, --script` | string | - | 脚本文件路径（支持通配符） |
| `-l, --language` | string | bash | 脚本语言类型 |
| `-p, --param` | []string | - | 脚本参数，格式：key=value |
| `-t, --timeout` | int | 300 | 执行超时时间（秒） |
| `-n, --parallel` | int | 1 | 并行执行数量 |
| `-o, --output` | string | - | 结果输出目录 |
| `-f, --format` | string | json | 输出格式 |
| `-w, --weight` | float64 | 1.0 | 脚本权重 |
| `-v, --verbose` | bool | false | 详细输出模式 |
| `-q, --quiet` | bool | false | 静默模式 |

```go
// BashAdapter Bash脚本适配器
type BashAdapter struct {
    apiScriptPath string // API脚本路径
    tempDir       string // 临时目录
}

// Execute 执行Bash脚本
func (a *BashAdapter) Execute(ctx context.Context, task *Task) (*ExecutionResult, error) {
    // 1. 准备执行环境
    env := a.prepareEnv(task)
    
    // 2. 构建执行命令
    // source api.sh && source script.sh
    cmd := a.buildCommand(task)
    
    // 3. 设置管道捕获输出
    stdout, stderr := a.setupPipes(cmd)
    
    // 4. 执行命令
    if err := cmd.Start(); err != nil {
        return nil, err
    }
    
    // 5. 等待执行完成或超时
    done := make(chan error, 1)
    go func() {
        done <- cmd.Wait()
    }()
    
    select {
    case <-ctx.Done():
        cmd.Process.Kill()
        return a.handleTimeout(task)
    case err := <-done:
        return a.handleResult(task, stdout, stderr, err)
    }
}
```

#### 5.6.4 归一化处理

```go
// Normalize 归一化处理 (Min-Max归一化)
func (p *ResultProcessor) Normalize(scores []float64) []float64 {
    if len(scores) == 0 {
        return scores
    }
    
    min, max := scores[0], scores[0]
    for _, s := range scores {
        if s < min {
            min = s
        }
        if s > max {
            max = s
        }
    }
    
    // 避免除零
    if max == min {
        normalized := make([]float64, len(scores))
        for i := range normalized {
            normalized[i] = 0.5
        }
        return normalized
    }
    
    // Min-Max归一化到[0, 1]
    normalized := make([]float64, len(scores))
    for i, s := range scores {
        normalized[i] = (s - min) / (max - min)
    }
    
    return normalized
}
```

```bash
    local level="$1"
    local message="$2"
    local timestamp
    timestamp=$(date '+%Y-%m-%d %H:%M:%S')

    local prefix="[$timestamp] [$level]"

}
```

```python
            import sys
```

```python
#!/usr/bin/env python3
# scripts/python/kb_api.py - 结果API部分

import json
import os
import time
from dataclasses import dataclass, field, asdict
from datetime import datetime
from enum import Enum
from typing import Any, Dict, List, Optional

class StepStatus(Enum):
    SUCCESS = "success"
    FAILURE = "failure"
    WARNING = "warning"
    SKIPPED = "skipped"

@dataclass
class StepResult:
    name: str
    status: str = "running"
    message: str = ""
    output: str = ""
    duration_ms: int = 0
    results: Dict[str, Any] = field(default_factory=dict)

@dataclass
class ScriptResult:
    script_name: str
    status: str = "running"
    steps: List[Dict] = field(default_factory=list)
    results: Dict[str, Any] = field(default_factory=dict)
    score: float = 0.0
    message: str = ""

class KBResult:
    def __init__(self):
        self.result_file = os.environ.get("KB_RESULT_FILE", "/tmp/kb_result.json")
        self.script_name = os.environ.get("KB_SCRIPT_NAME", "unknown")
        self.current_step: Optional[StepResult] = None
        self.result = ScriptResult(script_name=self.script_name)
        self._save_result()
    
    def _save_result(self) -> None:
        with open(self.result_file, "w", encoding="utf-8") as f:
            json.dump(asdict(self.result), f, indent=2, ensure_ascii=False)
    
    def step_start(self, name: str) -> None:
        if self.current_step is not None:
            self.step_warning("步骤未正确关闭")
        
        self.current_step = StepResult(name=name)
        self._save_result()
    
    def _end_step(self, status: StepStatus, message: str = "") -> None:
        if self.current_step is None:
            return
        
        self.current_step.status = status.value
        self.current_step.message = message
        self.result.steps.append(asdict(self.current_step))
        self.current_step = None
        self._save_result()
    
    def step_success(self, message: str = "执行成功") -> None:
        self._end_step(StepStatus.SUCCESS, message)
    
    def step_failure(self, message: str = "执行失败") -> None:
        self._end_step(StepStatus.FAILURE, message)
    
    def step_warning(self, message: str = "执行警告") -> None:
        self._end_step(StepStatus.WARNING, message)
    
    def step_skip(self, message: str = "跳过执行") -> None:
        self._end_step(StepStatus.SKIPPED, message)
    
    def step_output(self, output: str) -> None:
        if self.current_step:
            self.current_step.output = output
    
    def result(self, key: str, value: Any) -> None:
        self.result.results[key] = value
        self._save_result()
    
    def save(self) -> None:
        if self.current_step is not None:
            self.step_warning("步骤未正确关闭")
        
        # 计算总体状态和得分
        failure_count = sum(1 for s in self.result.steps if s["status"] == "failure")
        warning_count = sum(1 for s in self.result.steps if s["status"] == "warning")
        
        if failure_count > 0:
            self.result.status = "failure"
            self.result.score = 0.0
        elif warning_count > 0:
            self.result.status = "warning"
            self.result.score = max(0.0, 1.0 - warning_count * 0.1)
        else:
            self.result.status = "success"
            self.result.score = 1.0
        
        self._save_result()
```

### 8.2 Worker Pool实现

```go
// Scheduler 任务调度器
type Scheduler struct {
    taskQueue   chan *Task
    resultQueue chan *ExecutionResult
    workerPool  chan struct{}
    adapter     Adapter
    wg          sync.WaitGroup
    ctx         context.Context
    cancel      context.CancelFunc
}

// NewScheduler 创建调度器
func NewScheduler(maxWorkers int, adapter Adapter) *Scheduler {
    ctx, cancel := context.WithCancel(context.Background())
    return &Scheduler{
        taskQueue:   make(chan *Task, 100),
        resultQueue: make(chan *ExecutionResult, 100),
        workerPool:  make(chan struct{}, maxWorkers),
        adapter:     adapter,
        ctx:         ctx,
        cancel:      cancel,
    }
}

// Start 启动调度器
func (s *Scheduler) Start() {
    for i := 0; i < cap(s.workerPool); i++ {
        s.workerPool <- struct{}{}
    }
    
    go s.dispatch()
}

// dispatch 分发任务
func (s *Scheduler) dispatch() {
    for {
        select {
        case <-s.ctx.Done():
            return
        case task := <-s.taskQueue:
            <-s.workerPool
            s.wg.Add(1)
            go s.executeTask(task)
        }
    }
}

// executeTask 执行任务
func (s *Scheduler) executeTask(task *Task) {
    defer func() {
        s.workerPool <- struct{}{}
        s.wg.Done()
    }()
    
    ctx, cancel := context.WithTimeout(s.ctx, task.Timeout)
    defer cancel()
    
    result, err := s.adapter.Execute(ctx, task)
    if err != nil {
        result = &ExecutionResult{
            TaskID:       task.ID,
            Status:       StatusFailure,
            ErrorMessage: err.Error(),
        }
    }
    
    s.resultQueue <- result
}

// Submit 提交任务
func (s *Scheduler) Submit(task *Task) error {
    select {
    case s.taskQueue <- task:
        return nil
    default:
        return errors.New("task queue is full")
    }
}
```

```go
    requestID := getRequestID(r)
    
    resp := ErrorResponse{
        Success:   false,
        Timestamp: time.Now().Format(time.RFC3339),
        RequestID: requestID,
        Error: &ErrorDetail{
            Code:       err.Code,
            Category:   string(err.Category),
            Level:      string(err.Level),
            Message:    err.Message,
            Detail:     err.Detail,
            Suggestion: err.Suggestion,
            Retryable:  err.Retryable,
        },
    }

    w.Header().Set("X-Request-ID", requestID)
    w.Header().Set("X-Error-Code", fmt.Sprintf("%d", err.Code))

}
```

### 9.5 错误处理中间件

```go
package middleware

import (
    "context"
    "fmt"
    "net/http"
    "runtime/debug"
    "time"
    
    "kb-runnerx/internal/api/apiutil"
    "kb-runnerx/pkg/errors"
    "kb-runnerx/pkg/logger"
)

func ErrorRecovery(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        defer func() {
            if rec := recover(); rec != nil {
                stack := string(debug.Stack())
                
                logger.Error("panic recovered",
                    "path", r.URL.Path,
                    "method", r.Method,
                    "panic", fmt.Sprintf("%v", rec),
                    "stack", stack,
                )
                
                err := errors.NewError(
                    errors.ErrSystemPanic,
                    fmt.Sprintf("系统内部错误: %v", rec),
                    nil,
                )
                err.Detail = stack
                
                apiutil.WriteError(w, r, err)
            }
        }()
        
        next.ServeHTTP(w, r)
    })
}

func ErrorHandler(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        ctx := context.WithValue(r.Context(), "startTime", time.Now())
        r = r.WithContext(ctx)
        
        ww := &responseWriter{ResponseWriter: w}
        
        defer func() {
            if ww.statusCode >= 400 {
                logger.Warn("request failed",
                    "path", r.URL.Path,
                    "method", r.Method,
                    "status", ww.statusCode,
                    "duration", time.Since(ctx.Value("startTime").(time.Time)),
                )
            }
        }()
        
        next.ServeHTTP(ww, r)
    })
}

type responseWriter struct {
    http.ResponseWriter
    statusCode int
}

func (w *responseWriter) WriteHeader(statusCode int) {
    w.statusCode = statusCode
    w.ResponseWriter.WriteHeader(statusCode)
}
```

### 9.7 前端错误处理指南

```typescript
// 前端错误处理示例 (TypeScript)

interface ErrorResponse {
  success: false;
  error: {
    code: number;
    category: string;
    level: 'critical' | 'error' | 'warning' | 'info';
    message: string;
    detail?: string;
    suggestion?: string;
    retryable: boolean;
  };
  timestamp: string;
  request_id: string;
}

async function handleApiError(error: ErrorResponse) {
  const { error: err } = error;
  
  // 根据错误级别处理
  switch (err.level) {
    case 'critical':
      // 致命错误，显示错误页面
      showErrorPage({
        title: '系统错误',
        message: err.message,
        suggestion: err.suggestion,
        requestId: error.request_id,
      });
      break;
      
    case 'error':
      // 业务错误，显示错误提示
      if (err.retryable) {
        showRetryDialog({
          message: err.message,
          suggestion: err.suggestion,
          onRetry: () => retryLastRequest(),
        });
      } else {
        showErrorMessage(err.message, err.suggestion);
      }
      break;
      
    case 'warning':
      // 警告，显示提示但不阻断流程
      showWarningMessage(err.message);
      break;
      
    case 'info':
      // 信息提示
      showInfoMessage(err.message);
      break;
  }
  
  // 根据错误类别记录日志
  console.error(`[${err.category}] [${err.code}] ${err.message}`, {
    detail: err.detail,
    requestId: error.request_id,
  });
}

// 错误码判断工具函数
function isRetryableError(code: number): boolean {
  const retryableCodes = [
    1002, // 系统资源不足
    2000, // 执行引擎繁忙
    2001, // 执行超时
    2003, // 任务队列已满
    4004, // 脚本执行超时
    4005, // 脚本执行失败
    7000, // 后端连接失败
    7001, // 后端通信超时
    7002, // 后端响应错误
  ];
  return retryableCodes.includes(code);
}

function needsAuthRefresh(code: number): boolean {
  return code === 7003; // 后端认证失败
}

function isSystemError(code: number): boolean {
  return code >= 1000 && code < 2000;
}
```

## 附录B：脚本示例

### B.1 Bash脚本示例

```bash
#!/bin/bash
source ./scripts/bash/api.sh

kb_init

step "check_file_permissions"
if [ -r "/etc/passwd" ]; then
    result "readable" "true"
    step_success "File is readable"
else
    step_failure "File is not readable"
fi

step "check_disk_space"
df -h / | tail -1 | awk '{print $5}' | read usage
result "usage" "$usage"
if [ ${usage%\%} -lt 80 ]; then
    step_success "Disk usage is normal: $usage"
else
    step_warning "Disk usage is high: $usage"
fi

kb_save
```

### B.2 Python脚本示例

```python
#!/usr/bin/env python3
from kb_api import kb

with kb.step("check_process") as s:
    import subprocess
    result = subprocess.run(["ps", "aux"], capture_output=True, text=True)
    s.result("process_count", len(result.stdout.split('\n')))
    kb.step_success(f"Found {len(result.stdout.split(chr(10)))} processes")

with kb.step("check_memory") as s:
    import psutil
    memory = psutil.virtual_memory()
    s.result("total", memory.total)
    s.result("available", memory.available)
    s.result("percent", memory.percent)
    if memory.percent < 80:
        kb.step_success(f"Memory usage: {memory.percent}%")
    else:
        kb.step_warning(f"High memory usage: {memory.percent}%")

kb.save()
```
